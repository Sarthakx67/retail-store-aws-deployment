        - name: MYSQL_PASSWORD
          valueFrom:
            secretKeyRef:
              name: mysql-secret
              key: MYSQL_PASSWORD

          mountPath: /var/lib/mysql

          limits:
            cpu: 500m
            memory: 1Gi
          requests:
            cpu: 200m
            memory: 512Mi

# In Kubernetes, StatefulSets must use volumeClaimTemplates so that each replica gets its own persistent volume.
# Manually attaching a single PVC breaks scalability and reliability.

  volumeClaimTemplates:
  - metadata:
      name: mysql-data
    spec:
      accessModes:
        - ReadWriteOnce
      storageClassName: "ebs-sc"
      resources:
        requests:
          storage: 5Gi

# Yes, volumeClaimTemplates use dynamic EBS provisioning.
# Each StatefulSet pod gets its own EBS volume.
# Data is NOT shared automatically.
# Consistency is handled by the database using replication or clustering mechanisms, not by Kubernetes.

  volumeClaimTemplates:
  - metadata:
      name: postgresql-data
    spec:
      accessModes:
        - ReadWriteOnce
      storageClassName: "ebs-sc" # storageClassName must be quoted (Helm bug)
      resources:
        requests:
          storage: 5Gi
---
# Source: retail-store/charts/retail-store-rabbitmq/templates/statefullset.yaml
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: rabbitmq
  labels:
    tier: db
    environment: dev
spec:
  selector:
    matchLabels:
      app: rabbitmq # has to match .spec.template.metadata.labels
  serviceName: "rabbitmq"
  replicas: 1
  template:
    metadata:
      labels:
        app: rabbitmq # has to match .spec.selector.matchLabels
    spec:
      containers:
      - name: rabbitmq
        image: rabbitmq:3
        ports:
        - containerPort: 5672
          name: rabbitmq
        env:
        - name: RABBITMQ_DEFAULT_USER
          value: rabbitmq_user
        - name: RABBITMQ_DEFAULT_PASS
          valueFrom:
            secretKeyRef:
              name: rabbitmq-secret
              key: RABBITMQ_PASSWORD
---

# Source: retail-store/charts/retail-store-cart/templates/cart-servicemonitor.yaml
apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  name: cart-monitor
  namespace: monitoring
  labels:
    release: monitoring
spec:
  selector:
    matchLabels:
      app: cart
      component: backend
  namespaceSelector:
    matchNames:
      - retail-store-prod
  endpoints:

    path: /actuator/prometheus
    interval: 15s
